Aufgabe 1

Was gibt folgendes Programm auf dem Bildschirm aus?

#include <stdio.h>

      void inc( int *p) { ++*p; }

     

      int main(){

      int a[] = { 10, 20 }, i = 0;

      printf("*Demo mit Zeigern als Parameter*\n");

      inc(a);

      inc(&i);

      inc(&a[i]);

      inc(a+i);

      printf("\nErgebnis: i = %2d\n"

           " a[0] = %2d\n"

           " a[1] = %2d\n", i, a[0], a[1]);

      return 0;

      }

Was gibt folgendes Programm auf dem Bildschirm aus?

#include <stdio.h>

 

      int arr[] = { -20, -10, 0, 10, 20 };

      int main(){

      int *p, i;

      puts("*Demo mit Zeigern und Vektoren*\n");

      p = arr;

      while( p < arr+5)

           printf("%6d", *p++);

      putchar('\n');

      while( p-- > arr)

           printf("%6d", *p);

      putchar('\n');

      for( p=arr+2, i=0; i < 3; ++i)

           printf("%6d%6d\n", p[-i], p[i]);

      putchar('\n');

      return 0;

      }
