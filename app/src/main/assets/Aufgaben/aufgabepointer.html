Aufgabe 1

Was gibt folgendes Programm auf dem Bildschirm aus?

#include <stdio.h>

      void inc( int *p) { ++*p; }

     

      int main(){

      int a[] = { 10, 20 }, i = 0;

      printf("*Demo mit Zeigern als Parameter*\n");

      inc(a);

      inc(&i);

      inc(&a[i]);

      inc(a+i);

      printf("\nErgebnis: i = %2d\n"

           " a[0] = %2d\n"

           " a[1] = %2d\n", i, a[0], a[1]);

      return 0;

      }
